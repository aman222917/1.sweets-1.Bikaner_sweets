        // PASTE YOUR PUBLISHED CSV URL HERE

        const SHEET_URL = 'https://docs.google.com/spreadsheets/d/e/2PACX-1vRDlfQXzEpY7D42rWB37ByNmnAv8cEXOCDxRlwd5s7cyZg5nYHOuQUKMTmEt6SEaV1BLVU_SW3V9q_5/pub?gid=0&single=true&output=csv';

        const WHATSAPP_NUMBER = '91XXXXXXXXXX'; // Format: CountryCodeNumber

        let allProducts = [];

        let fuse;

        async function fetchData() {

            try {

                const response = await fetch(SHEET_URL);

                const data = await response.text();

                const rows = data.split('\n').slice(1);

               

                allProducts = rows.map(row => {

                    const cols = row.split(',');

                    return {

                        category: cols[0]?.trim(),

                        name: cols[1]?.trim(),

                        desc: cols[2]?.trim(),

                        originalPrice: cols[3]?.trim(),

                        discountPrice: cols[4]?.trim(),

                        unit: cols[5]?.trim(),

                        image: cols[6]?.trim(),

                        isHero: cols[7]?.toLowerCase().trim() === 'yes',

                        isSale: cols[8]?.toLowerCase().trim() === 'yes',

                        active: cols[9]?.toLowerCase().trim() === 'yes',

                        priority: parseInt(cols[10]) || 99

                    };

                }).filter(p => p.active && p.name);

                allProducts.sort((a, b) => a.priority - b.priority);

               

                // Initialize Smart Search

                fuse = new Fuse(allProducts, { keys: ['name', 'category'], threshold: 0.4 });

               

                renderUI(allProducts);

                renderCategories();

            } catch (err) { console.error("Error fetching sheet:", err); }

        }

        function createProductCard(p, isCompact = false) {

            const hasDiscount = p.originalPrice && p.originalPrice > p.discountPrice;

            const waMsg = `Hi, I want to order ${p.name} (${p.discountPrice}/${p.unit})`;

            const waLink = `[link removed]{WHATSAPP_NUMBER}?text=${encodeURIComponent(waMsg)}`;

            return `

                <div class="bg-white rounded-2xl shadow-sm border border-gray-100 overflow-hidden flex-shrink-0 ${isCompact ? 'w-48' : 'w-full'}">

                    <div class="relative h-40">

                        <img src="${p.image}" class="w-full h-full object-cover" alt="${p.name}">

                        ${hasDiscount ? `<span class="absolute top-2 left-2 sale-badge">SAVE</span>` : ''}

                    </div>

                    <div class="p-3">

                        <h3 class="font-bold text-sm truncate">${p.name}</h3>

                        <div class="flex items-center gap-2 mt-1">

                            <span class="text-orange-600 font-bold">₹${p.discountPrice}</span>

                            ${hasDiscount ? `<span class="text-gray-400 text-xs line-through">₹${p.originalPrice}</span>` : ''}

                        </div>

                        <p class="text-[10px] text-gray-400">${p.unit}</p>

                        <a href="${waLink}" class="mt-3 block text-center bg-green-500 text-white py-2 rounded-lg text-sm font-bold">

                            <i class="fab fa-whatsapp"></i> Order

                        </a>

                    </div>

                </div>

            `;

        }

        function renderUI(products) {

            const grid = document.getElementById('product-grid');

            const heroRail = document.getElementById('hero-rail');

            const saleRail = document.getElementById('sale-rail');

            // Hero Section

            const heroItems = products.filter(p => p.isHero).slice(0, 5);

            if (heroItems.length > 0) {

                document.getElementById('hero-section').classList.remove('hidden');

                heroRail.innerHTML = heroItems.map(p => `

                    <div class="relative w-80 h-48 rounded-2xl overflow-hidden flex-shrink-0 snap-center">

                        <img src="${p.image}" class="w-full h-full object-cover">

                        <div class="absolute inset-0 hero-gradient flex flex-col justify-end p-4 text-white">

                            <h3 class="text-lg font-bold">${p.name}</h3>

                            <p class="text-sm text-gray-200">Just ₹${p.discountPrice} / ${p.unit}</p>

                        </div>

                    </div>

                `).join('');

            }

            // Sale Section

            const saleItems = products.filter(p => p.isSale);

            if (saleItems.length > 0) {

                document.getElementById('sale-section').classList.remove('hidden');

                saleRail.innerHTML = saleItems.map(p => createProductCard(p, true)).join('');

            }

            // Main Grid

            grid.innerHTML = products.map(p => createProductCard(p)).join('');

        }

        function renderCategories() {

            const tabs = document.getElementById('category-tabs');

            const cats = ['All', ...new Set(allProducts.map(p => p.category))];

            tabs.innerHTML = cats.map(c => `

                <button onclick="filterCategory('${c}')" class="whitespace-nowrap px-4 py-2 rounded-full border border-gray-200 bg-white text-gray-600 text-sm font-medium">

                    ${c}

                </button>

            `).join('');

        }

        function filterCategory(cat) {

            const filtered = cat === 'All' ? allProducts : allProducts.filter(p => p.category === cat);

            renderUI(filtered);

        }

        document.getElementById('search-bar').addEventListener('input', (e) => {

            const query = e.target.value;

            if (query.length > 2) {

                const results = fuse.search(query).map(r => r.item);

                renderUI(results);

            } else if (query.length === 0) {

                renderUI(allProducts);

            }

        });

        fetchData();

    </script>
